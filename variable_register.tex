\section{Variables in Registers}

%Integer registers
\count0=42
The value is now `\the\count0'.

\def\macro{-123456}
\count0=\macro
The value is now `\the\count0'.

% The expansion expands everything follows
\def\firstmacro{123}
\def\secondmacro{456}
\def\thirdmacro{789}
\count0=\firstmacro\secondmacro\thirdmacro
The value is now `\the\count0'

% Abortion and Absorption
\count0=42
39 will not appear in this `\the\count0',
\def\macro{42}
\count0=\macro
39 unfortunately appears in `\the\count0'


%\relax is to end scanning
\count0=\macro\relax
39 will again not appear in this `\the\count0'

---------------------

%fixed point number registers

\dimen0=1pt
The value is now \the\dimen0.

\dimen0=0.0001pt
The value is now \the\dimen0.

\def\macro{1234.5678}
\dimen0=\macro pt
The value is now \the\dimen0.

--------------------------

\toks0={abc}
\toks1={DEF}
The value is now \the\toks0\the\toks1
