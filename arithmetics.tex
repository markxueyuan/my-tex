 \section{Arithmetics}

 % advance
 \newcount\tempa
 \tempa=40
 \advance\tempa by 10
 The value now is \the\tempa
 
 \newdimen\variable
 \variable=12pt
 \advance\variable by 10pt
 The value now is \the\variable
 
 --------------------------
 %multiply
 
 \multiply\tempa by -10
 The value now is \the\tempa
 
 \multiply\variable by 10
 The value now is \the\variable
